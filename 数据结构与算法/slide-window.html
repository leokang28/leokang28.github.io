function lengthOfLongestSubstring(s) {
    const window = new Map()
    let left = 0 
    let right = 0
    let res = 0
    while(right < s.length) {
        const c = s[right]
        right += 1
        if (window.has(c)) {
            window.set(c, window.get(c) + 1)
        } else {
            window.set(c, 1)
        }

        while(window.get(c) > 1) {
            const d = s[left]
            left += 1

            window.set(d, window.get(d) - 1)
        }

        res = Math.max(res, right - left)
    }
    return res
}
